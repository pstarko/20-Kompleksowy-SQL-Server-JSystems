/*
1. Wyświetl nazwiska i roczne dochody pracowników(salary + commission_pct). 
Posortuj wyniki po last_name (107 wierszy)
*/
select
	last_name,
	(salary + (salary * isnull(commission_pct,0))) * 12 roczna_placa
	from employees
order by last_name

/*
2. Wyświetl całość informacji o departamentach sortując wynik według nazw (27 wierszy)
*/
select
	*
	from departments
order by department_name

/*
3. Wyświetl listę job_id bez duplikatów(z tabeli employees) bez departamentu 20 (17 wierszy)
*/
select
	distinct job_id
	from employees
where
	department_id <> 20

select
	distinct job_id
	from employees
where
	department_id not in (20)
/*
4. wyświetl wszystkie informacje pracownikach na etatach AC_ACCOUNT i FI_ACCOUNT i 
posortuj ich według daty zatrudnienia(malejąco) (6 wierszy)
*/

select
	*
	from employees
where
	job_id in ('AC_ACCOUNT', 'FI_ACCOUNT')
order by
	hire_date desc
