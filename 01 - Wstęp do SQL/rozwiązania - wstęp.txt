/*
1. Wyświetl nazwiska i roczne dochody pracowników(salary + commission_pct). 
Posortuj wyniki po last_name (107 wierszy)
*/
select
	last_name,
	(salary + (salary * isnull(commission_pct,0))) * 12 roczna_placa
	from employees
order by last_name

/*
2. Wyświetl całość informacji o departamentach sortując wynik według nazw (27 wierszy)
*/
select
	*
	from departments
order by department_name

/*
3. Wyświetl listę job_id bez duplikatów(z tabeli employees) bez departamentu 20 (17 wierszy)
*/
select
	distinct job_id
	from employees
where
	department_id <> 20

select
	distinct job_id
	from employees
where
	department_id not in (20)
/*
4. wyświetl wszystkie informacje pracownikach na etatach AC_ACCOUNT i FI_ACCOUNT i 
posortuj ich według daty zatrudnienia(malejąco) (6 wierszy)
*/

select
	*
	from employees
where
	job_id in ('AC_ACCOUNT', 'FI_ACCOUNT')
order by
	hire_date desc


/*
5. wybierz tylko employee_id, last_name i job_id z departamentów 30,40 
w kolejności rosnących podstawowych zarobków (7 wierszy)
*/
select
	employee_id,
	last_name,
	job_id
	from employees
where
	department_id in (30, 40)
order by salary


/*
6. Pokaż nazwiska i płace podstawowe(salary) tych, 
których płace podstawowe nie mieszczą się w przedziale 2500 do 15000 (8 wierszy)
*/

select
	last_name,
	salary
	from employees
where
	salary not between 2500 and 15000

/*
7. wyświetl nazwisko, etat(job_id) i numer departamentu tych których nazwisko 
kończy się na „an” (4 wiersze)
*/
select
	last_name,
	job_id,
	department_id
	from employees
where
	last_name like '%an'

/*
8. wyświetl nazwisko i płace podstawową tych którzy zarabiają powyżej 10000 i mają szefa (14 wierszy)
*/

select
	last_name,
	salary
	from employees
where
	salary > 10000
	and manager_id is not null

select
	last_name,
	salary
	from employees
where
	salary > 10000
	and manager_id > 0

/*
9. wyświetl nazwiska, etaty i stawki godzinowe tych pracowników, 
którzy nie pracują jako AD_VP, PU_CLERK i SA_MAN i którzy zarabiają w przedziale od 5000 do 20000. 
Wyniki uszereguj według stawek godzinowych pracowników
(przyjmij 20 dniowy miesiąc pracy i 8 godzinny dzień pracy) 
(tylko płaca podstawowa) (50 wierszy)
*/

select
	last_name,
	job_id,
	salary / 20 /8 stawka_godzinowa
	from employees
where
	job_id not in ('AD_VP', 'PU_CLERK', 'SA_MAN')
	and salary between 5000 and 20000
order by
	stawka_godzinowa
 

/*
12. Czy są pracownicy, którzy w ciągu 5 lat spłacą kawalerkę, która kosztowała 250 000 wiedząc, 
że mogą odłożyć tylko ćwiartkę ze swojej pełnej miesięcznej pensji? 
Pokaż ich imiona, nazwiska pensję miesięczną i ile mogą miesięcznie odkładać (5 wierszy)
*/

select
	first_name,
	last_name,
	salary + (salary * isnull(commission_pct,0)) pensja,
	(salary + (salary * isnull(commission_pct,0))) * 0.25 
	from employees
where
	(salary + (salary * isnull(commission_pct,0))) * 0.25 * 5 * 12 > 250000
