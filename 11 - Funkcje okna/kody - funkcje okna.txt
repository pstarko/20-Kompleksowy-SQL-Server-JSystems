FROM employees

SELECT
	last_name,
	hire_date,
	department_id,
	LEAD(last_name,1,'----') OVER (partition by department_id 	ORDER BY hire_date) AS nastepny
FROM employees;

SELECT
	department_id,
	STRING_AGG(last_name,', ') nazwiska
from employees
group by department_id

SELECT
	job_id,
	STRING_AGG(last_name,',') within group (order by last_name) nazwiska
from employees
group by job_id

    ID,
    TRIM(value) AS opis

SELECT
	distinct department_id,
	FIRST_VALUE(last_name) over (partition by department_id order by salary DESC) last_name
from employees;

SELECT
	distinct department_id,
	FIRST_VALUE(last_name) over (partition by department_id order by salary) last_name
from employees;



SELECT
	distinct department_id,
	LAST_VALUE(last_name) over (partition by department_id order by salary DESC) last_name
from employees;

SELECT
	distinct department_id,
	LAST_VALUE(last_name) over (partition by department_id order by salary DESC
									rows between unbounded preceding
											 and unbounded following) last_name
from employees;

select 
	distinct department_id,
	sum(salary) over (partition by department_id) suma
from employees

select
		hire_date,
		salary,
		sum(salary) over (order by hire_date
							ROWS BETWEEN 1 PRECEDING 
							 		 AND 1 FOLLOWING) grow
from employees

select
	hire_date,
	salary,
	sum(salary) over (order by hire_date
						ROWS BETWEEN UNBOUNDED PRECEDING
								 AND CURRENT ROW) suma_kroczaca
from employees

select
	hire_date,
	salary,
	sum(salary) over (order by hire_date) suma_kroczaca
from employees

select
	hire_date,
	salary,
	sum(salary) over (order by hire_date
						RANGE BETWEEN UNBOUNDED PRECEDING
								 AND CURRENT ROW) suma_kroczaca
from employees


select
		hire_date,
		salary,
		sum(salary) over (order by hire_date
							ROWS BETWEEN 2 PRECEDING 
							 		 AND 1 PRECEDING) grow
from employees

select
	department_id,
	sum(salary),
	sum(sum(salary)) over (order by department_id
								ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) okno
from employees
	group by
	department_id
