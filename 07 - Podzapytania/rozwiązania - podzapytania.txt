/*
1. Wyświetl nazwiska i stanowiska pracowników pracujących w tym samym dziale co pracownik o nazwisku Yang (3 wiersze)
*/

select
	first_name,
	last_name
	from employees
where
	department_id = (select department_id from employees
						where last_name = 'Yang')


/*
2.Wyświetl następujące dane o najdłużej zatrudnionym pracowniku na stanowisku FI_ACCOUNT (1 wiersz)
*/

select
	*
	from employees
where
	hire_date = (select min(hire_date) from employees
					where job_id = 'FI_ACCOUNT')
	and job_id = 'FI_ACCOUNT'

/*
3. Zobacz najdłużej pracujących pracowników każdego działu. (12 wierszy)
*/

select
	first_name,
	last_name,
	department_id,
	hire_date
	from employees e
where
	hire_date = (select min(hire_date) from employees
					where department_id = e.department_id)



/*
4. Wyświetlaj te nazwy stanowisk, w których nie zatrudniono nikogo z działu MARKETING (17 wierszy)
*/

select
	*
	from jobs
