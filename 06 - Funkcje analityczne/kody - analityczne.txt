select 
    department_id,
    sum(salary)
    from employees
where
    department_id in (10,20,30)
group by rollup(department_id)

select 
    department_id,
    sum(salary)
    from employees
where
    department_id in (10,20,30)
group by cube(department_id)


select 
    department_id,
    max(salary)
    from employees
where
    department_id in (10,20,30)
group by rollup(department_id)

select 
    department_id,
    job_id,
    sum(salary) sum
    from employees
where department_id in (10,20,30)
group by rollup (department_id, job_id)

select 
    department_id,
    job_id,
    sum(salary) sum
    from employees
where department_id in (10,20,30)
group by cube (department_id, job_id)

select 
    department_id,
    job_id,
    sum(salary) sum,
    grouping(department_id) grouping_dept,
    grouping(job_id) grouping_job
    from employees
where department_id in (10,20,30)
group by rollup(department_id, job_id)


select 
    department_id,
    job_id,
    sum(salary) sum,
    grouping(department_id) grouping_dept,
    grouping(job_id) grouping_job
    from employees
where department_id in (10,20,30)
    or department_id is null
group by cube(department_id, job_id)

select 
    department_id,
    job_id,
    sum(salary) sum,
    grouping_id(department_id,job_id) grouping
    from employees
where department_id in (10,20,30)
group by cube(department_id,job_id)


select 
    department_id,
    job_id,
    sum(salary) sum,
    grouping_id(department_id,job_id) grouping
    from employees
where department_id in (10,20,30)
group by cube(department_id,job_id)
having grouping_id(department_id,job_id) = 3

select 
    department_id,
    job_id,
    sum(salary) sum
    from employees
where department_id in (10,20,30)
group by grouping sets((department_id), (department_id, job_id), (job_id))

select
