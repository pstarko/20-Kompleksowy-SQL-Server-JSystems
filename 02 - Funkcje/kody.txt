    last_name,
    salary,
    CASE
        WHEN salary > 10000 THEN 'OK'
        ELSE
            '---'
    END AS warunek

    employees

    last_name,
    salary,
    CASE
        WHEN salary > 20000 THEN '> 20k'
        WHEN salary > 10000 THEN '> 10k'
        ELSE
            '---'
    END AS warunek
FROM
    employees

--daty i czas
SELECT
    hire_date,
    year(hire_date) rok,
    month(hire_date) miesiac,
    day(hire_date) dzien
from employees


SELECT
    hire_date,
    getdate(),
    DATEDIFF(year, hire_date, getdate()) result_year,
    DATEDIFF(month, hire_date, getdate()) result_month,
    DATEDIFF(day, hire_date, getdate()) result_day
from employees

SELECT
    hire_date,
    DATEADD(year, 2, hire_date) result_year,
    DATEADD(month, 2,hire_date) result_month,
    DATEADD(day, 2, hire_date) result_day
from employees
