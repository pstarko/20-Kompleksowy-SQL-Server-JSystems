/*
1. Dla każdego pracownika wygeneruj kod składający się z pierwszych dwóch liter job_id 
oraz całego employee_id
*/

select
	last_name,
	job_id,
	employee_id,
	concat(left(job_id, 2), employee_id) code
	from employees


/*
2. Pozamieniaj litery l, m oraz k i zamień je na X. 
Zarówno poprzez funkcję TRANSLATE jak i REPLACE
*/

select
	last_name,
	translate(last_name, 'lmk','XXX') _translate,
	replace(
		replace(
			replace(last_name, 'l','X')
								, 'm','X')
									, 'k','X') _replace
	from employees



/*
3. Dodaj podwyżkę 15% dla każdego pracownika. Uwzględnij również dodatek(commission_pct). 
Posortuj dane po kolumnie z dodaną podwyżką, którą zamień na typ danych money.
*/

select
	last_name,
	cast((salary + (salary * isnull(commission_pct,0))) * 1.15 as money) more_money
	from employees
order by more_money


/*
4. Policz ile dni jest zatrudniony każdy pracownik. Wyróżnij odpowiednim komentarzem tych, 
którzy pracują ponad 3000 dni (Wynik na dzień 2024-01-01). Posortuj dane po nazwisku.
*/

select
	last_name,
	hire_date,
	concat('Pracuje ', DATEDIFF(day, hire_date, '2024-01-01')) result
	from employees
order by last_name


/*
5. Rozwiąż wyrażenie algebraiczne: Resztę z dzielenia 134 przez 8 przemnóż przez pierwiastek z 81. 
Podziel to wszystko przez 244 i podnieś do 4 potęgi. Do wyniku dodaj 0.0097. 
Zaogrąglij wszystko do 4 miejsc po przecinku zgodnie z zasadami matematyki.
*/

select
	ROUND(POWER(((134%8) * sqrt(81)) / 244, 4) + 0.0097, 4) result


/*
6. Wiedząc, że do otrzymania nagrody trzeba pracować 10 lat pokaż dane: 
datę zatrudnienia, ile lat pracy, ile lat do nagrody oraz informacje o tym, że ktoś już odebrał nagrodę. 
Jeśli komuś do nagrody brakuje mniej niż 2 lata napisz: „jeszcze troszeczkę!” – 
(Stan na 2025-02-02)
*/

select
	last_name,
	hire_date,
	datediff(year, hire_date, getdate()) ilosc_lat_przepracowanych,
	DATEADD(year, 10, hire_date) rok_nagrody,
	10 - datediff(year, hire_date, getdate()) ile_lat_do_nagrody,
	CASE 
		WHEN 10 - datediff(year, hire_date, getdate()) < 0 THEN 'Nagroda odebrana'
		WHEN 10 - datediff(year, hire_date, getdate()) > 1 THEN concat('Jeszcze ', 10 - datediff(year, hire_date, getdate()) , ' lat')
		WHEN 10 - datediff(year, hire_date, getdate()) >= 0 THEN 'Jeszcze troszeczkę'
		ELSE ' ---' 
	END result		
	from employees


/*
9. Pokaż last name w taki sposób, by ostatnie trzy litery były w odwróconej kolejności. 
Jeśli nazwisko ma mniej niż 3 litery odwróć całe nazwisko
*/

select
	last_name,
	case
		when LEN(last_name) < 3 then reverse(last_name)		
		else substring(last_name, 1, LEN(last_name) - 3) + reverse(right(last_name, 3))
	end result
	from employees
