/*
1. Dla każdego pracownika wygeneruj kod składający się z pierwszych dwóch liter job_id 
oraz całego employee_id
*/

select
	last_name,
	job_id,
	employee_id,
	concat(left(job_id, 2), employee_id) code
	from employees


/*
2. Pozamieniaj litery l, m oraz k i zamień je na X. 
Zarówno poprzez funkcję TRANSLATE jak i REPLACE
*/

select
	last_name,
	translate(last_name, 'lmk','XXX') _translate,
	replace(
		replace(
			replace(last_name, 'l','X')
								, 'm','X')
									, 'k','X') _replace
	from employees



/*
3. Dodaj podwyżkę 15% dla każdego pracownika. Uwzględnij również dodatek(commission_pct). 
Posortuj dane po kolumnie z dodaną podwyżką, którą zamień na typ danych money.
*/

select
	last_name,
	cast((salary + (salary * isnull(commission_pct,0))) * 1.15 as money) more_money
	from employees
order by more_money
