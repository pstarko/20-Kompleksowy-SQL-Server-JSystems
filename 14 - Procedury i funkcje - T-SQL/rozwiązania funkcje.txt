/*
1. (Tabela Employees) 
Użytkownik podaje stawkę(Rate). 
W wyniku pokazujesz ilość ludzi, którzy mają stawkę niższą, taką samą i wyższą od podanej. 
Wykorzystaj funkcję zwracającą tabelę.
*/

CREATE OR ALTER FUNCTION zadanie_1 (@stawka int)
returns table
as
RETURN
	select
		(select count(*) from employees where rate > @stawka) wyzsza,
		(select count(*) from employees where rate = @stawka) taka_sama,
		(select count(*) from employees where rate < @stawka) nizsza
GO

select * from dbo.zadanie_1(30)

/*
2. (Tabela Employees) 
Stwórz funkcję która pokaże imię i nazwisko pracowników ze wskazanego departamentu. 
Jeśli nie ma pracowników również pokaż stostowną informację.
Wykorzystaj Multi statement table valued function.
*/

CREATE OR ALTER FUNCTION zadanie_2 (@dept int)
returns 
	@tab1 table (fn varchar(50), ln varchar(50))
as
begin
	insert into @tab1 
	select
		firstname,
		LastName
	from Employees
		where departmentid = @dept
	return
end

select * from dbo.zadanie_2(1)
