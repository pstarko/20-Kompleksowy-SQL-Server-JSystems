create or alter procedure insert_to_numery (@ilosc int)
as
declare @i int = 1
begin
	--drop table #numery
	create table numery (id int)
	while (@i <= @ilosc)
		begin
			insert into numery values (@i)
			set @i = @i + 1
		end
	select * from numery
end

--KURSOR

DECLARE @EmployeeID INT, @FirstName NVARCHAR(50), @LastName NVARCHAR(50)

-- Deklaracja kursora
DECLARE employee_cursor 
	CURSOR FOR
			SELECT employee_id, first_name, last_name
			FROM EMPLOYEES

	-- Otwarcie kursora
	OPEN employee_cursor

		-- Pobranie pierwszego rekordu
		FETCH NEXT FROM employee_cursor INTO @EmployeeID, @FirstName, @LastName

		-- Iteracja po rekordach
			WHILE @@FETCH_STATUS = 0
			BEGIN
				PRINT 'ID: ' + CAST(@EmployeeID AS NVARCHAR(10)) + ', Imię: ' + @FirstName + ', Nazwisko: ' + @LastName

				-- Pobranie kolejnego rekordu
				FETCH NEXT FROM employee_cursor INTO @EmployeeID, @FirstName, @LastName
			END

	-- Zamknięcie i usunięcie kursora
	CLOSE employee_cursor
	DEALLOCATE employee_cursor
GO
