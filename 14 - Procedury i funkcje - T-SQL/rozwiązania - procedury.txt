/*
1. Stwórz procedurę, która pozwoli zmienić status w kolumnie EmailPromotion 
wybranego przez id pracownika. 
Status nie może być zamieniony na niższy. 

Dany pracownik może mieć zmieniony status tylko raz przez procedurę. 
Znajdź rozwiązanie, które Ci to umożliwi

1. Tworzymy tabelkę AllData_log(id int)
2. Wyciągamy EmailPromotion z tabeli AllData
3. Sprawdzamy czy EP=2, jeśli tak - "nie możemy go zmienić"
4. Sprawdzamy ile mamy wpisów dla id w AllData_log
5. Jeśli więcej niż 0 to "nie możemy zmieniać"
6. W innym wypadku możemy zmieniać
*/


create table AllData_log(id int)

go

create or alter procedure Zmiana @id int
as
begin
declare @email int
	select
		@email = EmailPromotion from AllData 
		where id = @id
	if @email = 2
		begin
			select 'nie mogę zmienić EmailPromotion'
		end
	else
		begin
			if (select count(*) from AllData_log where id = @id) = 0
				begin
					update AllData set EmailPromotion = EmailPromotion + 1 where id = @id
					insert into AllData_log values (@id)
					select 'zmieniłem EmailPromotion'
				end
			else
				begin
					select concat('Pracownik o id = ', @id, ' miał już zmieniony EmailPromotion')
				end
		end
end

exec Zmiana 1
