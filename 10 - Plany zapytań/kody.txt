	soh.SalesOrderID, soh.OrderDate, soh.CustomerID,
	sod.ProductID, sod.OrderQty, sod.UnitPrice
FROM Sales.SalesOrderHeader AS soh
JOIN Sales.SalesOrderDetail  AS sod ON soh.SalesOrderID = sod.SalesOrderID

CREATE OR ALTER FUNCTION dbo.GetLangName (@langid int)
RETURNS nvarchar(100)
AS 
BEGIN
	RETURN (SELECT TOP 1 alias FROM sys.syslanguages WHERE msglangid = @langid)
END
GO

SELECT language_id, dbo.GetLangName(language_id) AS language, COUNT(*)
FROM msgs
GROUP BY language_id
