select
	max(salary),
	min(salary),
	avg(salary),
	sum(salary),
	count(*)
	from employees

select
	count(*),
	count(1),
	count('dzik jest dziki'),
	count(commission_pct),
	count(coalesce(commission_pct,99))
	from employees

select
	*
	from employees
where
	1=1 and
	department_id = 30

	--SELECT * FROM [employees] WHERE [department_id]=@1

	--select * from employees where 1=1 and department_id = 30

select 
    count(department_id) dept, 
    count(distinct department_id) dist_dept
    from employees

select 
    department_id,
    Sum(salary) suma
    from employees
group by department_id 

select 
    department_id,
    job_id,
    Sum(salary) suma
    from employees
group by department_id, job_id 

select 
    department_id,
    Sum(salary) suma
    from employees
group by department_id
having Sum(salary) > 10000

select 
    department_id,
    Sum(salary) suma
    from employees
where Sum(salary) > 10000
group by department_id

    Sum(salary) suma
    from employees
