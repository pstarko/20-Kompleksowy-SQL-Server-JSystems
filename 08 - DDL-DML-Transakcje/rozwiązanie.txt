1. Tabela Sklep (idsklepu(PK), miasto)
2. Tabela Pracownik (idpracownika(PK), nazwisko, sklepid(FK z tabeli sklep))
3. Dodaj po kilka wierszy do obu tabel.
4. Włącz transakcję i spróbuj użyć innej sesji do dodania danych
5. Użyj rollbacka
*/

-- 1. Tworzenie tabeli Sklep
CREATE TABLE Sklep (
    idsklepu INT PRIMARY KEY,
    miasto VARCHAR(50)
);

-- 2. Tworzenie tabeli Pracownik
CREATE TABLE Pracownik (
    idpracownika INT PRIMARY KEY,
    nazwisko VARCHAR(50),
    sklepid INT FOREIGN KEY REFERENCES Sklep(idsklepu)
);

/*
6. Dołóż trzecią tabelę z wynagrodzeniami(id(PK), pracownikid(FK), sklepid(FK), wynagrodzenie)
7. Nie pozwól by w wynagrodzeniu moża było wpisywać wartość mniejszą niż 3000
8. Dodaj kilka wierszy do wynagrodzeń
*/
-- 6. Tworzenie tabeli Wynagrodzenie
CREATE TABLE Wynagrodzenie (
    id INT PRIMARY KEY,
    pracownikid INT FOREIGN KEY REFERENCES Pracownik(idpracownika),
    sklepid INT FOREIGN KEY REFERENCES Sklep(idsklepu),
    wynagrodzenie DECIMAL(10,2) CONSTRAINT CHK_wynagrodzenie CHECK (wynagrodzenie >= 3000)
);
