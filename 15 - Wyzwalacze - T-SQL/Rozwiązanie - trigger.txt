CREATE OR ALTER TRIGGER trg_Zamowiania
ON Zamowienia
AFTER INSERT
AS
BEGIN
	declare @idproduktu int, @iloscZamowiona int, @iloscDostepna int

    SELECT 
        @IdProduktu = IdProduktu,
        @IloscZamowiona = IloscZamowiona
    FROM INSERTED;

    SELECT 
        @IloscDostepna = IloscDostepna
    FROM Produkty
    WHERE IdProduktu = @IdProduktu;

    IF @iloscZamowiona > @iloscDostepna
        BEGIN
            RAISERROR('Brak wystarczającej ilości', 16, 1)
            ROLLBACK
        END
    ELSE
        update Produkty set IloscDostepna = IloscDostepna - @iloscZamowiona where IdProduktu = @idproduktu
END
