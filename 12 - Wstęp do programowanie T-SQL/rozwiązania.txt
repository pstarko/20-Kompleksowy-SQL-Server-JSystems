/*
1.Użytkownik podaje dwie liczby X i Y. 
Następnie pokazujesz mu liczby całkowite od X do Y.
*/

declare @x int = 0, @y int = 10
begin
	while @x <= @y
		begin
			print @x
			set @x = @x + 1
		end
end


/*
2.
Stwórz pętlę pokazującą informację: 
Liczba 50, Liczba 49…., Liczba 1. 

Pętla rozpoczyna się od 50 a kończy na 1. 
Pomiń liczby parzyste pomiędzy 20 a 30.
*/

declare @x int = 50, @y int = 1
begin
	while @x >= @y
		begin
				if @x between 20 and 30 and @x % 2 = 1
					print concat('z between: ', @x)	
				else if @x not between 20 and 30
						print @x

		set @x = @x - 1
		end
end

select * from klienci

DECLARE @output NVARCHAR(MAX) = '';
begin
    UPDATE Klienci
    SET Status = CASE
        WHEN Saldo > 10000 THEN 'VIP'
        WHEN Saldo BETWEEN 1000 AND 10000 THEN 'Standard'
        WHEN Saldo < 1000 THEN 'Zagrożony'
    END;

    SELECT @output = @output + 
        'Id: ' + CAST(Id AS NVARCHAR) + 
        ' | ' + Imie + ' ' + Nazwisko + 
        ' | Saldo: ' + CAST(Saldo AS NVARCHAR) + 
        ' | Status: ' + Status + CHAR(13)
    FROM Klienci
    ORDER BY Id;

    PRINT @output;
end


/*
5. Napisz program do obliczania kolejnych liczb całkowitych. Np. po podaniu liczby 5 otrzymasz 15(czyli 1+2+3+4+5)
*/

declare @x int = 1, @wynik int = 0, @koncowa_wartosc int = 6
begin
	while @x <= @koncowa_wartosc
		begin
			set @wynik = @wynik + @x
			print concat('Dodałem: ', @x, ' = ' , @wynik)
			set @x = @x + 1
		end
	print Concat('Końcowy wynik: ' ,@wynik)
end


/*
6.
Napisz program, który zwróci Ci liczbę liter i cyfr w podanych ciągu znaków i cyfr. 
Znajdź wcześniej odpowiednią funkcję, która Ci to umożliwi
*/

--select ISNUMERIC(1), isnumeric('a')

declare
	@ciagZnakow varchar(50) = 'Agent00766',
	@liczbaLiter int = 0,
	@liczbaCyfr int = 0,
	@indeks int = 1
begin
	while @indeks <= len(@ciagZnakow)
		begin
			if ISNUMERIC(SUBSTRING(@ciagZnakow, @indeks,1)) = 1
				set @liczbaCyfr = @liczbaCyfr + 1
			else
				set @liczbaLiter = @liczbaLiter + 1

		set @indeks = @indeks + 1
		end
	print concat('Ilosc liter: ', @liczbaLiter)
	print concat('Ilosc cyfr: ', @liczbaCyfr)
end
