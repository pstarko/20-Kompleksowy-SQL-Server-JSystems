/*
1. Pokaż imię, nazwisko, numer działu i stanowisko dla wszystkich pracowników (106 wierszy)
*/

select
	e.first_name,
	e.last_name,
	e.department_id,
	j.job_title
	from employees e join jobs j on e.job_id = j.job_id
					 join departments d on e.department_id = d.department_id


/*
2. Pokaż pracowników pracujących w Londynie i Toronto(3 wiersze)
*/

select
	e.first_name,
	e.last_name,
	l.city
	from employees e JOIN departments d on e.department_id = d.department_id
					 JOIN locations l on d.location_id = l.location_id
where
	l.city in ('London','Toronto')

/*
3. Pokaż poniższy raport nazwisk i nazw departamentów.
Uwzględnij departamenty, które nie zatrudniają pracowników. 
Nie bierz pod uwagę zespołu Operations(88 wierszy)
*/

select
	e.last_name,
	d.department_name
	from employees e RIGHT JOIN departments d on e.department_id = d.department_id
where
	d.department_name not in ('Operations')


/*
4. Pokaż ilość zatrudnionych pracowników w Stretford, Bombay, Sydney, London, Munich i Seattle(6 wierszy)
*/

select
	l.city,
	count(e.employee_id) ilosc_pracownikow
	from employees e JOIN departments d on e.department_id = d.department_id
					 right JOIN locations l on l.location_id = d.location_id
where
	l.city in ('Stretford', 'Bombay', 'Sydney', 'London', 'Munich', 'Seattle')
group by 
	l.city
