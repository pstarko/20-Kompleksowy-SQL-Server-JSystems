select 
	e.first_name,
	e.last_name,
	e.salary,
	l.street_address,
	l.city,
	rank() over (order by e.salary desc) rnk
	from employees e join departments d on e.department_id=d.department_id
					 join locations l on d.location_id=l.location_id

		e.first_name,

		l.street_address,

		rank() over (order by e.salary desc) rnk,
		dense_rank() over (order by e.salary desc) drnk
		from employees e join departments d on e.department_id=d.department_id
						 join locations l on d.location_id=l.location_id ) t1

	t1.rnk  <= 5

/*
5. Policz pracowników dla każdego departamentu i dodaj ranking procentowy (użyj funkcji
PERCENT_RANK).

6. Zmodyfikuj zapytanie z punktu poprzedniego w taki sposób, aby otrzymać informacje o
departamentach, które lokują się w 25% najliczniej obsadzonych departamentów.
*/


select
	d.department_name,
	COUNT(*) counter,
	percent_rank() over (order by count(*) desc) * 100 prk
	from employees e join departments d on e.department_id=d.department_id
group by 
	d.department_name


select
	t1.*,
	percent_rank() over (order by counter desc) * 100 prk
	from (
			select
				d.department_name,
				COUNT(*) counter
				from employees e join departments d on e.department_id=d.department_id
			group by 
				d.department_name ) t1

--6
select
	t1.*
	from (
			select
				d.department_name,
				COUNT(*) counter,
				percent_rank() over (order by count(*) desc) * 100 prk
				from employees e join departments d on e.department_id=d.department_id
			group by 
				d.department_name ) t1
where
	t1.prk <= 25
