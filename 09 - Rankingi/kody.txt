--Rankingi

SELECT
    first_name,
    last_name,
    salary,
    row_number() over (order by salary) row_number
FROM
    employees


SELECT
    first_name,
    last_name,
    salary,
    department_id,
    row_number() over (partition by department_id order by salary) row_number
FROM
    employees


SELECT
    first_name,
    last_name,
    salary,
    rank() over (order by salary) rank,
    dense_rank() over (order by salary) drank
FROM
    employees

select 
    t1. val,
    CUME_DIST() over (order by t1.val),
    PERCENT_RANK() over (order by t1.val)
from(
    SELECT
        100 as val  
    union
    SELECT
        200 as val  
    union
    SELECT
        300 as val  
    union
    SELECT
        600 as val ) t1

SELECT
    first_name,
    last_name,
    job_id,
    salary,
    ntile(4) over (order by salary ) ntile
FROM
    employees
where job_id in ('FI_ACCOUNT', 'IT_PROG')
